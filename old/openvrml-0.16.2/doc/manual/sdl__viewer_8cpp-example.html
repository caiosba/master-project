00167             virtual const std::string do_type() const throw ()
00168             {
00169                 //
00170                 // A real application should use OS facilities for this.  This
00171                 // is a crude hack because sdl-viewer uses std::filebuf in
00172                 // order to remain simple and portable.
00173                 //
00174                 using std::find;
00175                 using std::string;
00176                 using boost::algorithm::iequals;
00177                 using boost::next;
00178                 string media_type = "application/octet-stream";
00179                 const string::const_reverse_iterator dot_pos =
00180                     find(this->url_.rbegin(), this->url_.rend(), '.');
00181                 if (dot_pos == this->url_.rend()
00182                     || next(dot_pos.base()) == this->url_.end()) {
00183                     return media_type;
00184                 }
00185                 const string::const_iterator hash_pos =
00186                     find(next(dot_pos.base()), this->url_.end(), '#');
00187                 const string ext(dot_pos.base(), hash_pos);
00188                 if (iequals(ext, "wrl")) {
00189                     media_type = "model/vrml";
00190                 } else if (iequals(ext, "x3dv")) {
00191                     media_type = "model/x3d+vrml";
00192                 } else if (iequals(ext, "png")) {
00193                     media_type = "image/png";
00194                 } else if (iequals(ext, "jpg") || iequals(ext, "jpeg")) {
00195                     media_type = "image/jpeg";
00196                 }
00197                 return media_type;
00198             }
00199 
00200             virtual bool do_data_available() const throw ()
00201             {
00202                 return !!(*this);
00203             }
00204         };
00205 
00206         const string scheme = uri.substr(0, uri.find_first_of(':'));
00207         if (scheme != "file") {
00208             throw invalid_argument('\"' + scheme + "\" URI scheme not "
00209                                    "supported");
00210         }
00211         //
00212         // file://

00264         bool need_update = false;
00265         bool need_redraw = false;
00266         do {
00267             SDL_Event event;
00268             sdl_viewer::event_info viewer_event_info;
00269             SDL_WaitEvent(0);
00270             while (SDL_PollEvent(&event)) {
00271                 switch (event.type) {
00272                 case SDL_VIDEOEXPOSE:
00273                     this->post_redraw();
00274                     break;
00275                 case SDL_VIDEORESIZE:
00276                     SDL_SetVideoMode(event.resize.w,
00277                                      event.resize.h,
00278                                      0,
00279                                      sdl_viewer::video_mode_flags);
00280                     this->resize(event.resize.w, event.resize.h);
00281                     this->post_redraw();
00282                     break;
00283                 case SDL_KEYDOWN:
00284                     viewer_event_info.event = sdl_viewer::event_key_down;
00285                     switch (event.key.keysym.sym) {
00286                     case SDLK_HOME:
00287                         viewer_event_info.what = sdl_viewer::key_home;
00288                         break;
00289                     case SDLK_LEFT:
00290                         viewer_event_info.what = sdl_viewer::key_left;
00291                         break;
00292                     case SDLK_UP:
00293                         viewer_event_info.what = sdl_viewer::key_up;
00294                         break;
00295                     case SDLK_RIGHT:
00296                         viewer_event_info.what = sdl_viewer::key_right;
00297                         break;
00298                     case SDLK_DOWN:
00299                         viewer_event_info.what = sdl_viewer::key_down;
00300                         break;
00301                     case SDLK_PAGEDOWN:
00302                         viewer_event_info.what = sdl_viewer::key_page_down;
00303                         break;
00304                     case SDLK_PAGEUP:
00305                         viewer_event_info.what = sdl_viewer::key_page_up;
00306                         break;
00307                     default:
00308                         break;
00309                     }
00310                     this->input(&viewer_event_info);
00311                     break;
00312                 case SDL_MOUSEBUTTONDOWN: case SDL_MOUSEBUTTONUP:
00313                     viewer_event_info.event = event.button.state == SDL_PRESSED
00314                                             ? sdl_viewer::event_mouse_click
00315                                             : sdl_viewer::event_mouse_release;
00316                     viewer_event_info.what = event.button.button - 1;
00317                     viewer_event_info.x = event.button.x;
00318                     viewer_event_info.y = event.button.y;
00319                     this->input(&viewer_event_info);
00320                     break;
00321                 case SDL_MOUSEMOTION:
00322                     if (!event.motion.state) {
00323                         viewer_event_info.event = sdl_viewer::event_mouse_move;
00324                         viewer_event_info.x = event.motion.x;
00325                         viewer_event_info.y = event.motion.y;
00326                         this->input(&viewer_event_info);
00327                     } else {
00328                         for (Uint8 button = SDL_BUTTON_LEFT;
00329                              button < 4;
00330                              ++button) {
00331                             if (event.motion.state & SDL_BUTTON(button)) {
00332                                 viewer_event_info.event =
00333                                     sdl_viewer::event_mouse_drag;
00334                                 viewer_event_info.what = button - 1;
00335                                 viewer_event_info.x = event.motion.x;
00336                                 viewer_event_info.y = event.motion.y;
00337                                 this->input(&viewer_event_info);
00338                             }
00339                         }
00340                     }
00341                     break;
00342                 case SDL_QUIT:
00343                     done = true;
00344                     break;
00345                 case SDL_USEREVENT:
00346                     switch (event.user.code) {
00347                     case redraw_event_code:
00348                         need_redraw = true;
00349                         break;
00350                     case update_event_code:
00351                         need_update = true;
00352                         break;
00353                     }
00354                     break;
00355                 default:
00356                     break;
00357                 }
00358             }
00359             if (need_update) {
00360                 this->update();
00361                 need_update = false;
00362             }
00363             if (need_redraw) {
00364                 this->redraw();
00365                 need_redraw = false;
00366             }
00367         } while (!done);
